Connect to Google Sheets
All data syncs to your own Google Sheet. Cloud backup. Accessible from any device.
STEP 1 — Create your Google Sheet
Go to sheets.google.com, create a new sheet named Master Financial Control. Copy the ID from the URL.
https://docs.google.com/spreadsheets/d/[COPY THIS PART]/edit
STEP 2 — Add the sync script (supports backup AND restore)
In your sheet: Extensions → Apps Script. Delete existing code, paste the script below, click Save, then Deploy → New Deployment → Web App. Set Execute as: Me and Who has access: Anyone. Click Deploy, authorize when prompted, and copy the Web App URL.
function doPost(e){ try{ var body=e.postData.contents; var d=JSON.parse(body); var ss=SpreadsheetApp.openById(d.sheetId); var sh=ss.getSheetByName(d.tab)||ss.insertSheet(d.tab); // Soft delete: set Status = DELETED for matching row ID if(d.updateStatus){ var id=String(d.updateStatus.id); var rows=sh.getDataRange().getValues(); for(var i=1;i<rows.length;i++){ if(String(rows[i][0])===id){ var col=rows[0].indexOf('Status')+1; if(col>0) sh.getRange(i+1,col).setValue(d.updateStatus.status); break; } } return ContentService.createTextOutput('{"status":"OK"}').setMimeType(ContentService.MimeType.JSON); } // Edit: update existing row in place by ID if(d.updateRow){ var id=String(d.updateRow.id); var rows=sh.getDataRange().getValues(); for(var i=1;i<rows.length;i++){ if(String(rows[i][0])===id){ sh.getRange(i+1,1,1,d.updateRow.row.length).setValues([d.updateRow.row]); break; } } return ContentService.createTextOutput('{"status":"OK"}').setMimeType(ContentService.MimeType.JSON); } if(d.clear)sh.clearContents(); if(d.headers&&sh.getLastRow()===0)sh.appendRow(d.headers); if(d.rows)d.rows.forEach(function(r){sh.appendRow(r);}); return ContentService.createTextOutput('{"status":"OK"}').setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService.createTextOutput('{"status":"ERROR","message":"'+err.toString()+'"}').setMimeType(ContentService.MimeType.JSON); } } function doGet(e){ try{ var ss=SpreadsheetApp.openById(e.parameter.sheetId); var tab=e.parameter.tab||'Master Log'; var sh=ss.getSheetByName(tab); if(!sh)return ContentService.createTextOutput('{"rows":[]}').setMimeType(ContentService.MimeType.JSON); var data=sh.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({rows:data})).setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService.createTextOutput('{"rows":[],"error":"'+err.toString()+'"}').setMimeType(ContentService.MimeType.JSON); } }
⚠ IMPORTANT — If you already deployed the old script
Replace the old code with the new script above, save, then go to Deploy → Manage Deployments → Edit (pencil icon) → Version: New Version → Deploy. Use the same Web App URL — no need to reconnect.
